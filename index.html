qui.quadrado<-function(x,cross=""){

if(cross=="f2d") return(chisq.test(table(x),p=c(1/4,3/4))$p.value)

else if(cross=="f2c") return(chisq.test(table(x),p=c(1/4,1/2,1/4))$p.value)

else stop("Tipo de cruzamento inválido") }

Dominant.results<-apply(C8.genotip,2,qui.quadrado,cross="f2d") Codominant.results<-apply(B3.genotip,2,qui.quadrado,cross="f2c")

LG1_final_map <- draw.map(LG1.final, names= TRUE, grid=TRUE, cex.mrk=0.6) setwd ("E:/ESALQ/PÓS GRADUAÇÃO_ESALQ_USP/DOUTORADO_2014/disciplinas/Biometria de Marcadores Gen/aula6") write.map(LG1.final, "mapa_onemap_G1.map")
